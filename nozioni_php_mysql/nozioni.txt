? CREAZIONE DATABASE IN PHP
! nome server
 $hostnome = '127.0.0.1';
! username
 $username = 'root';
! password
 $password = '';

! creare la connessione
 $conn = new mysqli($hostnome, $username, $password);

! controllo errore crazione
 if ($conn->connect_errno) {
 die("connessione fallita: " . $conn->connect_error);
 } else {
 echo "connesione stabilita";
 }

!creazione database
 $sql = 'CREATE DATABASE esercizi_php_mysql';
 if ($conn->query($sql) === true) {
 echo "database creato";
 } else {
 echo "errore nella creazione del database";
 }

! chiudi la connessione
 $conn->close();


? CREAZIONE DI UNA TABELLA  
<?php
! hostname
$hostname = '127.0.0.1';
! username
$username = 'root';
! password
$password = '';
! database
$databse = 'esercizi_php_mysql';

! crea la connessione
$conn = new mysqli($hostname, $username, $password, $databse);

! controllo connessione
if ($conn->errno) {
    die("connessione non riuscita" . $conn->error);
} else {
    echo "connessione avvenuta <br>";
}

! Crea la tabella 
$sql = 'CREATE TABLE dipendenti (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nome VARCHAR(30) NOT NULL,
    cognome VARCHAR(30) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    assunzione DATE DEFAULT CURRENT_DATE,
    ruolo VARCHAR(50) DEFAULT "non specificato"
)';

! controllo quary
if (!$conn->query($sql)) {
    echo "errore nella creazione";
} else {
    echo "tabella creata con successo <br>";
}

$conn->close();
